actor BaseWeaponSlot : CustomInventory
{
	Radius 20
	Height 16
	Inventory.PickupMessage ""
	Inventory.PickupSound ""
	+INVENTORY.UNDROPPABLE
	Args 0
	States
	{
		Spawn:
			TNT1 A 0
			TNT1 A 0 ACS_NamedExecuteAlways("WEAPONSTATE", 0, args[0])
			goto SpawnLoop
		SpawnLoop:
			"####" "#" 35
			"####" "#" 0 ACS_NamedExecuteAlways("WEAPONSTATE", 0, args[0])
			loop
		HideWeapon:
			TNT1 A 0
			goto SpawnLoop
		PP7:
			PIST A 0
			goto SpawnLoop
		AutomaticShotgun:
			SGN2 A 0
			goto SpawnLoop
		ZMG:
			ZMGP A 0
			goto SpawnLoop
		Klobb:
			MGUN A 0
			goto SpawnLoop
		KF7Soviet:
			KF7P A 0
			goto SpawnLoop
		GoldenGun:
			BFUG A 0
			goto SpawnLoop
		GE64Shotgun:
			SHOT A 0
			goto SpawnLoop
		GE64RocketLauncher:
			LAUN A 0
			goto SpawnLoop
		MoonrakerLaser:
			PLAS A 0
			goto SpawnLoop
		CougarMagnum:
			RAIL A 0
			goto SpawnLoop
		SniperRifle:
			SNPR A 0
			goto SpawnLoop
		GE64GrenadeLauncher:
			GRLN A 0
			goto SpawnLoop
		RCP90:
			RCPP A 0
			goto SpawnLoop
		DD44:
			DD4P A 0
			goto SpawnLoop
		AR33:
			AR3P A 0
			goto SpawnLoop
		ThrowingKnives:
			TKNP A 0
			goto SpawnLoop
		ProximityMines:
			AMMO A 0
			goto SpawnLoop
		RemoteMines:
			AMMO A 0
			goto SpawnLoop
		TimedMines:
			AMMO A 0
			goto SpawnLoop
	}
}
