actor BaseAmmoSlot : CustomInventory
{
	Radius 20
	Height 16
	Inventory.PickupMessage ""
	Inventory.PickupSound ""
	+INVENTORY.UNDROPPABLE
	Args 0
	States
	{
		Spawn:
			TNT1 A 0
			TNT1 A 0 ACS_NamedExecuteAlways("AMMOSTATE", 0, args[0])
			goto SpawnLoop
		SpawnLoop:
			CLIP A 35
			CLIP A 0 ACS_NamedExecuteAlways("AMMOSTATE", 0, args[0])
			loop
		HideAmmo:
			TNT1 A 35
			TNT1 A 0 ACS_NamedExecuteAlways("AMMOSTATE", 0, args[0])
			loop
	}
}
